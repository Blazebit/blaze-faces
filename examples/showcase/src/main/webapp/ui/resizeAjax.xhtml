Resizable - AJAX

When resizing is completed, a backing bean listener can be invoked via ajax behavior.

Source

<h:form>
    <b:growl id="growl" showDetail="true" />

    <b:graphicImage id="campnou" value="/images/barca/camp_nou.jpg" />

    <b:resizable for="campnou" animate="true" ghost="true">
        <b:ajax listener="\#{resizeBean.onResize}" update="growl" />
    </b:resizable>
</h:form>
                    
package com.blazebit.blazefaces.examples.view;

import javax.faces.application.FacesMessage;
import javax.faces.context.FacesContext;
import com.blazebit.blazefaces.event.ResizeEvent;

public class ResizeBean {

    public void onResize(ResizeEvent event) {
        FacesMessage msg = new FacesMessage(FacesMessage.SEVERITY_INFO,
                        "Image resized", "Width:" + event.getWidth() + ",Height:" + event.getHeight());

        FacesContext.getCurrentInstance().addMessage(null, msg);
    }
}
