Dialog - Basic

Dialog visibility is managed by using show() and hide() methods of the client side api.

Source

	
<h:panelGrid columns="1" cellpadding="5">
    <b:commandButton id="basic" value="Basic" onclick="dlg1.show();" type="button" />

    <b:commandButton id="modalDialogButton" value="Modal" onclick="dlg2.show();" type="button"/>

    <b:commandButton id="effectsDialogButton" value="Effects" onclick="dlg3.show();" type="button" />
</h:panelGrid>

<b:dialog id="basicDialog" header="Basic Dialog" widgetVar="dlg1">
    <h:outputText value="Resistance to BlazeFaces is futile!" />
</b:dialog>

<b:dialog id="modalDialog" header="Modal Dialog" widgetVar="dlg2" modal="true" height="100">
    <h:outputText value="This is a Modal Dialog." />
</b:dialog>

<b:dialog header="Effects" widgetVar="dlg3" showEffect="explode" hideEffect="bounce" height="100">
    <h:outputText value="This dialog has nice effects." />
</b:dialog>
