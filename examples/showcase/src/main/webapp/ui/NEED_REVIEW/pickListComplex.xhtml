PickList - Complex

PickList provides a dual list to switch items between each other. Since picklist is a composite component it uses a facet based design to make customization of controls easier.




Source Target


Source

<h:form id="form">
	<b:pickList value="\#{pickListBean.players}" var="player" 
				itemLabel="\#{player.name}" itemValue="\#{player}" converter="player">
		<f:facet name="add">
			<b:graphicImage value="/images/picklist/add.png" />
		</f:facet>
		<f:facet name="addAll">
			<b:graphicImage value="/images/picklist/addall.png" />
		</f:facet>
		<f:facet name="remove">
			<b:graphicImage value="/images/picklist/remove.png" />
		</f:facet>
		<f:facet name="removeAll">
			<b:graphicImage value="/images/picklist/removeall.png" />
		</f:facet>
	</b:pickList>

	<b:commandButton value="Submit" update="form:display"/>
	
	<b:outputPanel id="display">
		<table>
			<tr>
				<td><strong>Source</strong></td>
				<td><strong>Target</strong></td>
			</tr>
			
			<tr>
				<td valign="top">
					<ui:repeat value="\#{pickListBean.players.source}" var="item">
						<h:outputText value="\#{item.name}" />
						<br />
					</ui:repeat>
				</td>
				<td valign="top">
					<ui:repeat value="\#{pickListBean.players.target}" var="item">
						<h:outputText value="\#{item.name}" />
						<br />
					</ui:repeat>
				</td>
			</tr>
		</table>
	</b:outputPanel>
	
</h:form>
