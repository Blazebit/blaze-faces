Effects on Load

Effects can also be applied to a JSF component when page is loaded for the first time or after an ajax request is completed.

Source

<h:form>

	<b:messages id="messages">
		<b:effect type="pulsate" event="load" delay="1000">
			<f:param name="mode" value="'show'" />
		</b:effect>
	</b:messages>

	<b:panel header="Message Effects" toggleable="true">
		<h:panelGrid columns="2">
			<h:outputText value="Type: *" />
			<b:inputText value="\#{buttonBean.text}" required="true" label="Text"/>

			<h:outputText />
			<b:commandButton value="Save" update="messages"
					actionListener="\#{buttonBean.displayMessage}" />
		</h:panelGrid>
	</b:panel>

</h:form>
                    
public class ButtonBean {

	private String text;

	public String getText() {
		return text;
	}

	public void setText(String text) {
		this.text = text;
	}

	public void displayMessage(ActionEvent actionEvent) {
		FacesMessage message = new FacesMessage(FacesMessage.SEVERITY_INFO, "You said:'" + text + "'", null);
		FacesContext.getCurrentInstance().addMessage(null, message);
	}
}
