Resizable

Resizable features various configuration options.

Resize Me! Keep the aspect ratio. A ghost is displayed, don't be scared. Resize can be animated as well. I have min/max boundaries. Resize offset is 20 pixels. I can be resized to any direction. I'm restricted to my parent's boundaries.

Source

<b:panel id="basic" header="Basic Resizable">
    Resize Me!
</b:panel>
<b:resizable for="basic" />

<b:panel id="aspectRatio" header="Aspect Ratio">
    Keep the aspect ratio.
</b:panel>
<b:resizable for="aspectRatio" aspectRatio="true"/>

<b:panel id="ghost" header="Ghost">
    A ghost is displayed, don't be scared.
</b:panel>
<b:resizable for="ghost" ghost="true"/>

<b:panel id="animation" header="Animated">
    Resize can be animated as well.
</b:panel>
<b:resizable for="animation" animate="true" effect="swing" effectDuration="normal"/>

<b:panel id="boundaries" header="Boundaries" style="width:300px;height:150px;">
    I have min/max boundaries.
</b:panel>
<b:resizable for="boundaries" minWidth="200" maxWidth="400" minHeight="100" maxHeight="200" />

<b:panel id="grid" header="Grid">
    Resize offset is 20 pixels.
</b:panel>
<b:resizable for="grid" grid="20"/>

<b:panel id="handles" header="Handles">
    I can be resized to any direction.
</b:panel>
<b:resizable for="handles" handles="e,w,n,se,sw,ne,nw"/>

<b:outputPanel layout="block" style="width:400px;height:200px;border:2px solid black;">
    <b:panel id="containment" header="Containment" style="margin:0">
        I'm restricted to my parent's boundaries.
    </b:panel>
    <b:resizable for="containment" containment="true"/>
</b:outputPanel>
