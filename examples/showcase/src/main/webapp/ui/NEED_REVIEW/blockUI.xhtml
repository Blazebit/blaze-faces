BlockUI

BlockUI blocks other JSF Components. It can be used with special ajax integration or manually with client side api.

<h:form>
                
    <b:growl id="growl" />

    <h3>Basic BlockUI</h3>
    <b:panel id="pnl" header="New User" style="margin-bottom:10px;">
        <b:messages id="messages" />
        <h:panelGrid columns="3">
            <h:outputLabel for="firstname" value="Firstname: *" />
            <b:inputText id="firstname" value="\#{partialProcessRequestBean.firstname}" required="true" label="Firstname">
                <f:validateLength minimum="2" />
            </b:inputText>
            <b:message for="firstname" />

            <h:outputLabel for="surname" value="Surname: *" />
            <b:inputText id="surname" value="\#{partialProcessRequestBean.surname}" required="true" label="Surname"/>
            <b:message for="surname" />
        </h:panelGrid>

        <b:commandButton id="saveBtn" value="Save" icon="ui-icon-check" 
                    style="margin:0" actionListener="\#{partialProcessRequestBean.savePerson}" update="growl"/>
    </b:panel>

    <b:blockUI block="pnl" trigger="saveBtn" />

    <h3>Advanced BlockUI</h3>
    <b:dataTable id="dataTable" var="car" value="\#{tableBean.cars}"
                    paginator="true" rows="10">
        <f:facet name="header">
            Ajax Pagination
        </f:facet>

        <b:column>
            <f:facet name="header">
                <h:outputText value="Model" />
            </f:facet>
            <h:outputText value="\#{car.model}" />
        </b:column>

        <b:column>
            <f:facet name="header">
                <h:outputText value="Year" />
            </f:facet>
            <h:outputText value="\#{car.year}" />
        </b:column>

        <b:column>
            <f:facet name="header">
                <h:outputText value="Manufacturer" />
            </f:facet>
            <h:outputText value="\#{car.manufacturer}" />
        </b:column>

        <b:column>
            <f:facet name="header">
                <h:outputText value="Color" />
            </f:facet>
            <h:outputText value="\#{car.color}" />
        </b:column>
    </b:dataTable>

    <b:blockUI block="dataTable" trigger="dataTable">
        LOADING<br />
        <b:graphicImage value="/images/ajax-loader.gif"/>
    </b:blockUI>

    <h3>Client Side API</h3>
    <b:tree value="\#{treeBean.root}" var="node" id="tree">
        <b:treeNode id="treeNode">
            <h:outputText value="\#{node}" id="lblNode"/>
        </b:treeNode>
    </b:tree>

    <b:commandButton id="pnlBtn" value="Block Panel" type="button" onclick="bui.show()"/>
    <b:commandButton id="pnlBtn2" value="Unblock Panel" type="button" onclick="bui.hide()"/>

    <b:blockUI block="tree" widgetVar="bui"/>

</h:form>
