Draggable

Draggable can make any JSF component draggable.


Source

<b:panel id="pnl" header="Draggable Panel">
	<h:outputText value="I am actually a regular p:panel." />
</b:panel>
<b:draggable for="pnl" />

<b:panel id="handlepnl" header="Handle">
	<h:outputText value="I can only be dragged with my header." />
</b:panel>
<b:draggable for="handlepnl" handle=".ui-panel-titlebar" />

<b:panel id="hpnl" header="Horizontal Only">
	<h:outputText value="I can only be dragged horizontally." />
</b:panel>
<b:draggable for="hpnl" axis="x" />

<b:panel id="vpnl" header="Vertical Only">
	<h:outputText value="I can only be dragged vertically" />
</b:panel>
<b:draggable for="vpnl" axis="y" />

<b:panel id="cpnl" header="Clone">
	<h:outputText value="I am cloned in dragging" />
</b:panel>
<b:draggable for="cpnl" helper="clone" />

<b:panel id="rpnl" header="Revert">
	<h:outputText value="I will be reverted back to my original position" />
</b:panel>
<b:draggable for="rpnl" revert="true" />

<b:panel id="opnl" header="Opacity">
	<h:outputText value="My opacity is lower when I'm being dragged" />
</b:panel>
<b:draggable for="opnl" opacity="0.3" />

<b:panel id="gpnl" header="Grid">
	<h:outputText value="I am dragged in grid mode" />
</b:panel>
<b:draggable for="gpnl" grid="20,40" />

<b:outputPanel layout="block" style="width:400px;height:200px;border:1px solid #666666;">
	<b:panel id="conpnl" header="Restricted">
		<h:outputText value="I am restricted to my parent's boundaries" />
	</b:panel>
</b:outputPanel>
<b:draggable for="conpnl" containment="parent" />
