Watermark

Watermark displays a hint about input fields by using native placeholder in supported browsers and a javascript solution in others browser compatibility.

<h:form>

	<b:messages showDetail="true"/>

	<h:panelGrid columns="3">

		<h:outputLabel value="Search: "/>
		<b:inputText id="keyword" value="\#{watermarkBean.keyword}" required="true" label="Keyword" />
		<b:watermark for="keyword" value="Search with a keyword" />

	</h:panelGrid>

	<b:commandButton actionListener="\#{watermarkBean.search}" value="Regular" ajax="false" />
	<b:commandButton actionListener="\#{watermarkBean.search}" value="Ajax" onclick="BlazeFaces.cleanWatermarks();" oncomplete="BlazeFaces.showWatermarks();"  />

package com.blazebit.blazefaces.examples.view;

public class WatermarkBean {

	private String keyword;

	public String getKeyword() {
		return keyword;
	}

	public void setKeyword(String keyword) {
		this.keyword = keyword;
	}

	public void search() {
		FacesContext.getCurrentInstance().addMessage(null, new FacesMessage(FacesMessage.SEVERITY_INFO,"No results found with ", "'" + keyword + "'"));
	}
}
